class Wave:
    def __init__(self, A, k, omega, T=None, nu=None):
        self.A = A
        self.k = k
        self.omega = omega
        self.T = T if T else 2*pi/self.omega
        self.nu = nu if nu else 1/self.T
        
    def x(self, z, t):
        return self.A * sin(self.omega*t - self.k*z)
    
    def __str__(self):
        return f'Wave with A = {self.A}, k = {self.k} and omega = {self.omega}'

class StandWave(Wave):
    def x(self, z, t):
        return 2*self.A*cos(self.k*z)*sin(self.omega*t)

def makeData(func):
    x = np.arange (-10, 10, 0.05)
    y = np.arange (-10, 10, 0.05)
    xgrid, ygrid = np.meshgrid(x, y)
    zgrid = func(xgrid, ygrid)
    return xgrid, ygrid, zgrid
